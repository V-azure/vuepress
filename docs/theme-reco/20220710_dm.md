<template>
  <div class="dm">
    <div class="show">
      <span v-for="item in txt">{{ item }}</span>
    </div>
    <div class="send">
      <div class="input">
        <input
          v-model="inptxt"
          @keyup.enter="f"
          type="text"
          name=""
          id="inp"
          placeholder="请在此输入弹幕"
        />
      </div>
      <div class="bu" @click="f">发送</div>
    </div>
  </div>
</template>

<script>
import $ from "jquery";
export default {
  data() {
    return {
      txt: [],
      inptxt: "",
    };
  },
  methods: {
    f() {
      if (this.inptxt.trim() == "") {
        alert("输入的内容不能为空");
      } else {
        this.txt.push(this.inptxt);
      }
      this.inptxt = "";
      
    },
  },
};
</script>
