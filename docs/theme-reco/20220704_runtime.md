```javascript
// js   （vue文件放入mouted里，html文件把这些定义函数使用）
setInterval(() => {
        var X = new Date("6/26/2022 00:00:00");
        var Y = new Date();
        var T = (Y.getTime() - X.getTime());
        var M = 24 * 60 * 60 * 1000;
        var a = T / M;
        var A = Math.floor(a);
        var b = (a - A) * 24;
        var B = Math.floor(b);
        var c = (b - B) * 60;
        var C = Math.floor((b - B) * 60);
        var D = Math.floor((c - C) * 60);
        runtime.innerHTML = "☀️本站已运行🌙: " + A + "天⭐" + B + "小时⭐" + C + "分⭐" + D + "秒⭐"
      }, 1000);
```
